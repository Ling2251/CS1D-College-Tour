366/*
367 * recursiveCollegeSort()
368 * This function will creat the shortest way to visit different colleges.
369 * For example, from the starting campus, the next campus will be the closest to the startding college, then the closest after that.
370 * Each recursive call find the closest campus to the campus passed in.
371 * When the closest is found, the new campus is sorted and then used for the next call.
372 */
373void planTripWindow::recursiveCollegeSort(QString currentCamp)
374{
375 sortedCampuses.enqueue(currentCamp);
376 selectedCampuses.removeAll(currentCamp);
378 // first check to see if there is anly college in the selectedCampuse vector
379 if(selectedCampuses.isEmpty())
380 {
381 return;
382 }
383 else
384 {
385 // set the frist campises to a string
386 QString* otherCamp = selectedCampuses.begin();
387 int currentIndex = 0;
388
389 // getting the distanct using GetDistBtwn function in dbmanger
390 double leastDist = m_database.GetDistBtwn(currentCamp, *otherCamp);
391 int leastIndex = 0;
392
393 otherCamp++;
394 currentIndex++;
395
396 // keep looping thorgh the colleges untill it has reach to the end of the vector
397 while(otherCamp != selectedCampuses.end())
398 {
399 // // getting the current distanct using GetDistBtwn function in dbmanger
400 double currentDist = m_database.GetDistBtwn(currentCamp, *otherCamp);
401
402 // do the comparsion
403 if (currentDist < leastDist)
404 {
405 leastDist = currentDist;
406 leastIndex = currentIndex;
407 }
408 else
409 {
410 otherCamp++;
411 currentIndex++;
412 }
413 }
414 QString nextCamp = selectedCampuses.at(leastIndex);
415 recursiveCollegeSort(nextCamp);
416 }
417}

648double planTripWindow::calculateDistance(int selectedCampusCount)
649{
650 //initial a double type total
651 double distance = 0.0;
652
653 // set the campises to a string
654 QString* currentCampus = selectedCampuses.begin();
655 QString* otherCampus = selectedCampuses.begin();
656
657 // loop through for each campus that was selected
